## Task 2: Fix Cmd+Enter in Monaco

**Files:**
- Modify: `src/components/Editor.tsx`
- Modify: `src/App.tsx` (remove Enter binding from global useKeyBindings)

Monaco captures keyboard events internally. The fix is to use `editor.addAction()` which registers the shortcut inside Monaco's own keybinding system.

`src/components/Editor.tsx` — add `onMount` prop:

```tsx
import MonacoEditor, { OnMount } from '@monaco-editor/react';
import * as monaco from 'monaco-editor';

interface EditorProps {
  isDark: boolean;
  onRun: () => void;
}

export default function Editor({ isDark, onRun }: EditorProps) {
  // ...existing state...

  const handleMount: OnMount = (editor) => {
    // Register Cmd+Enter inside Monaco so it fires even when editor has focus
    editor.addAction({
      id: 'cp-run',
      label: 'Run Solution',
      keybindings: [monaco.KeyMod.CtrlCmd | monaco.KeyCode.Enter],
      run: () => onRun(),
    });
  };

  return (
    <div className="h-full overflow-hidden">
      <MonacoEditor
        height="100%"
        language="cpp"
        value={code}
        onChange={handleChange}
        onMount={handleMount}
        theme={isDark ? 'vs-dark' : 'vs'}
        options={{ /* same as before */ }}
      />
    </div>
  );
}
```

In `App.tsx`, pass `onRun={handleRun}` to `<Editor>` via the `<Layout>` and remove the `Cmd+Enter` entry from `useKeyBindings` (keep Cmd+Shift+S, Cmd+M, Cmd+T, Cmd+O).

type RunStatus = keyof typeof STATUS_MAP;

function getStatus(isCompiling: boolean, isRunning: boolean, result: import('../lib/types').RunResult | null): RunStatus {
  if (isCompiling) return 'compiling';
  if (isRunning)   return 'running';
  if (!result)     return 'ready';
  if (result.compile_errors.length > 0) return 'compile_error';

use std::sync::Mutex;
static RUNNING_PID: Mutex<Option<u32>> = Mutex::new(None);

// Inside run_solution, after spawn:
if let Ok(mut pid_guard) = RUNNING_PID.lock() {
    *pid_guard = Some(child.id());
}
// After completion, clear it:
if let Ok(mut pid_guard) = RUNNING_PID.lock() {
    *pid_guard = None;
}

pub fn stop_running_process() {
    if let Ok(mut pid_guard) = RUNNING_PID.lock() {
        if let Some(pid) = *pid_guard {
            unsafe { libc::kill(pid as i32, libc::SIGTERM); }
            *pid_guard = None;
        }
    }
}

import { useState, useEffect, useRef } from 'react';

  onNewProblem: () => void;
  onDataView: () => void;
}

export default function ProblemSidebar({ onOpenProblem, onNewProblem, onDataView }: ProblemSidebarProps) {
  const problems = useStore(s => s.problems);
  const currentProblem = useStore(s => s.currentProblem);
  const [search, setSearch] = useState('');
  const [focused, setFocused] = useState(0);
  const searchRef = useRef<HTMLInputElement>(null);

  const filtered = problems.filter(p =>
    p.name.toLowerCase().includes(search.toLowerCase()) ||
    (p.url ?? '').toLowerCase().includes(search.toLowerCase())
  );

  useEffect(() => { setFocused(0); }, [search]);

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'ArrowDown') { e.preventDefault(); setFocused(f => Math.min(f + 1, filtered.length - 1)); }
    if (e.key === 'ArrowUp')   { e.preventDefault(); setFocused(f => Math.max(f - 1, 0)); }
    if (e.key === 'Enter' && filtered[focused]) onOpenProblem(filtered[focused].id);
  };

  return (
    <div className="sidebar flex-col">
      <div className="panel-header" style={{ justifyContent: 'space-between', textTransform: 'none', fontSize: 12 }}>
        <span style={{ fontWeight: 600, letterSpacing: 0 }}>Problems</span>

      </div>

      <div style={{ padding: '8px 8px 4px' }}>
        <input
          ref={searchRef}
          className="input input-sm"
          placeholder="Search..."
          value={search}
          onChange={e => setSearch(e.target.value)}
          onKeyDown={handleKeyDown}
          style={{ width: '100%' }}

import { useState, useEffect } from 'react';
import { useStore } from '../store/useStore';
import { api } from '../lib/tauri';

import ConfirmDialog from './ConfirmDialog';

interface DataManagementProps {

export default function DataManagement({ onClose, onOpenProblem }: DataManagementProps) {

  const setProblems = useStore(s => s.setProblems);

      setProblems(updated);
      setRenaming(null);

                    className="input input-sm"
                    autoFocus
                    value={renameName}
                    onChange={e => setRenameName(e.target.value)}
                    onBlur={() => handleRename(p.id)}

      )}

# Replace make_icon() with:
def make_icon(size):
    s = size
    c = Canvas(s, s)
    sc = s / 512

    def S(v): return max(1, int(v * sc))

    BG     = (10,  15,  30, 255)   # #0a0f1e deep navy
    CARD   = (16,  24,  48, 255)   # #101830 card
    TEAL   = (0,  200, 180, 255)   # accent teal
    WHITE  = (230, 237, 243, 255)

    # Full background
    for y in range(s):
        for x in range(s):
            c.px[y][x] = BG

    # Rounded card
    margin = S(32)
    c.fill_rounded_rect(margin, margin, s-margin, s-margin, S(64), CARD)

    # Teal accent stripe bottom-left corner
    accent_w = S(120); accent_h = S(8)
    c.fill_rounded_rect(margin, s-margin-accent_h, margin+accent_w, s-margin, S(4), TEAL)

    # "C" letter — two horizontal bars + left vertical bar
    # Centered, large and bold
    cx = int(s * 0.35)  # center of C
    cy = int(s * 0.50)
    bar_h = S(28); bar_w = S(70); vert_h = S(110); vert_w = S(28)
